遍历 ​

下面提供几种遍历方式参考

swift
var a = [1: "张三", 2: "李四"]

// 键值对遍历
for (k, v) in a {
    print(k, v)
}

// 值遍历
for v in a.values {
    print(v)
}

// 键遍历
for k in a.keys {
    print(k)
}

// forEach 默认参数遍历
a.forEach{
    print($0, $1)
}

// forEach 定义参数遍历
a.forEach{k, v in
    print(k, v)
}
