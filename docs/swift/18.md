# 案例

## 水果列表切换

```swift
import SwiftUI

func myFood() -> String {
    return ["🍓", "🍒", "🍇", "🍉", "🍊", "🍐", "🍎", "🍌"].randomElement()!
}

struct ContentView: View {
    @State private var todat = myFood()

    var body: some View {
        VStack {
            Spacer() // 撑开
            Text("\(todat)") // html 框架
                .font(.system(size: 100)) // css 样式
                .padding()
            Spacer() // 撑开
            Button("今天吃啥？"){
                // 脚本
                todat = myFood()
            }
            .padding()
            .background(.teal)
            .foregroundColor(.white)
            .cornerRadius(30)
            Spacer() // 撑开
        }
        .padding()
    }
}

#Preview {
    ContentView()
}
```
